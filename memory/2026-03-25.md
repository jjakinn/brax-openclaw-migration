**Pattern Matching Layer:**
```python
def parse_command(user_input):

        (r'search\s+(.+)', 
         lambda m: ('web_search', {'query': m.group(1)})),
        (r'weather\s+(.+)', 
         lambda m: ('weather', {'city': m.group(1)})),
    ]
    # Returns (tool_name, params) or (None, None)
```

**Execution Layer:**
```python

    },
    # ... other tools
}

    # Uses hardcoded path, substitutes params

```

### Flow Diagram
```
User Input
    ↓

    ↓

    r"subprocess\.run\(\['/Users/Jakin/",
    "subprocess.run(['python3', '/Users/Jakin/",
    code

```

### Issue 3: Model Uses '3' Instead of 'python3'
**Symptom:** `subprocess.run(['3', '/path/to/tool.py'...])`

**Fix Applied:**
- Wrapper script `tools/3` that executes as python3
- Runtime correction: `code = code.replace("['3',", "['python3',")`

### Issue 4: Double Processing
**Symptom:** After executing code, model includes it in response, causing re-execution

**Fix Applied:**
```python
# Strip code blocks before storing in history
clean_response = re.sub(r'```python\n.*?\n```', '', response, flags=re.DOTALL)
self.history.append({'user': input, 'assistant': clean_response})
```

### Issue 5: Tool Names
**Symptom:** Model uses `youtube_ingest.py` instead of `knowledge_ingest.py`

**Fix Applied:**
- Symlink: `tools/youtube_ingest.py -> knowledge_ingest.py`

# HOW TO: Convert Any Ollama Cloud Model Into an Agent

## The Problem

When you run `ollama run <model>:cloud`, it opens an **interactive chat interface** that:
- Traps you in a conversation bubble
- Cannot execute commands on your system
- Says "I can't do that" when asked to perform actions
- Requires manual copy/paste of any generated code

## The Solution Architecture

Transform the model from a **chatbot** into an **agent** by adding:

```
┌─────────────────────────────────────────────────────────────┐
│  USER INPUT                                                  │
└─────────────────────────────────────────────────────────────┘
              ↓
┌─────────────────────────────────────────────────────────────┐
│  WRAPPER SCRIPT (Python)                                     │
│  • Intercepts your input                                     │
│  • Pattern matches common commands                           │
│  • Routes to tools or AI as needed                           │
└─────────────────────────────────────────────────────────────┘
              ↓
    ┌─────────┴─────────┐
    ↓                   ↓
┌──────────┐    ┌──────────────┐
│  TOOLS   │    │  AI FALLBACK │
│ (Direct) │    │ (Generate    │
│          │    │  & Execute)  │
└──────────┘    └──────────────┘
```

### Step 4: Create the Agent Wrapper

**agent.py** (the main wrapper):

```python
#!/usr/bin/env python3
"""
Agent Wrapper - Routes input to tools or AI
"""
import subprocess
import re
import sys
import json

AGENT_DIR = "/Users/<username>/.ollama/agents/<model>"
TOOLS_DIR = f"{AGENT_DIR}/tools"
MODEL_NAME = "<model>-agent"

# ═══════════════════════════════════════════════════════════════
# PATTERN MATCHER - Catches common commands
# ═══════════════════════════════════════════════════════════════
PATTERNS = [
    # High confidence (95%) - Specific patterns FIRST
    (r'open\s+up\s+a\s+(?:new\s+)?(?:youtube\s+)?tab\s+in\s+safari\s+and\s+search\s+(?:for\s+)?(.+)',
     'safari_youtube', {'query': 1}),
    (r'open\s+youtube\s+(?:and\s+)?search\s+(?:for\s+)?(.+)',
     'safari_youtube', {'query': 1}),
    (r'search\s+youtube\s+(?:for\s+)?(.+)',
     'safari_youtube', {'query': 1}),
    (r'open\s+(https?://\S+)', 'safari_open', {'url': 1}),
    (r'weather\s+(?:in\s+)?(.+)', 'weather', {'city': 1}),
    
    # Medium confidence (80%)
    (r'search\s+(.+)', 'web_search', {'query': 1}),
]

def match_pattern(user_input):
    """Match input to pattern"""
    text = user_input.lower().strip()
    
    for pattern, tool_name, param_map in PATTERNS:
        match = re.search(pattern, text)
        if match:
            params = {}
            for key, group_num in param_map.items():
                params[key] = match.group(group_num).strip()
            return tool_name, params
    
    return None, None

# ═══════════════════════════════════════════════════════════════
# TOOL EXECUTOR - Runs tools safely
# ═══════════════════════════════════════════════════════════════
TOOL_MAP = {
    'safari_youtube': ['python3', f'{TOOLS_DIR}/safari_control.py', '--youtube-search'],
    'safari_open': ['python3', f'{TOOLS_DIR}/safari_control.py', '--open'],
    'weather': ['python3', f'{TOOLS_DIR}/api_integration.py', '--weather'],
    'web_search': ['python3', f'{TOOLS_DIR}/web_search.py', '--search'],
}

def execute_tool(tool_name, params):
    """Execute a tool"""
    if tool_name not in TOOL_MAP:
        return False, f"Unknown tool: {tool_name}"
    
    cmd = TOOL_MAP[tool_name].copy()
    
    # Add parameters
    if tool_name == 'safari_youtube':
        cmd.append(params['query'])

#!/usr/bin/env python3
"""
ReAct Agent Loop - Plan → Execute → Observe → Iterate
"""
import subprocess
import json
import re
from typing import List, Dict

AGENT_DIR = "/Users/Jakin/.ollama/agents/qwen"
TOOLS_DIR = f"{AGENT_DIR}/tools"

class AgentLoop:
    """
    ReAct-style agent that plans and executes multi-step tasks
    """
    
    def __init__(self, model_name="qwen-agent"):
        self.model = model_name
        self.max_iterations = 10
    
    def run(self, task: str) -> str:
        """Main agent loop"""
        print(f"🎯 Task: {task}")
        print("=" * 50)
        
        # Step 1: Plan
        plan = self._create_plan(task)
        print(f"\n📋 Plan:")
        for i, step in enumerate(plan, 1):
            print(f"  {i}. {step}")
        
        # Step 2: Execute plan
        results = []
        for i, step in enumerate(plan, 1):
            print(f"\n🔹 Step {i}/{len(plan)}: {step}")
            
            result = self._execute_step(step, task, results)
            results.append({'step': step, 'result': result})
            
            if 'error' in result.lower():
                print(f"  ⚠️  Error encountered, replanning...")
                plan = self._replan(task, results)
                if plan:
                    print(f"  🔄 New plan created")
        
        # Step 3: Synthesize final answer
        final_answer = self._synthesize(task, results)
        
        return final_answer
    
    def _create_plan(self, task: str) -> List[str]:
        """Create execution plan"""
        # For simple tasks, create plan locally
        # For complex tasks, ask AI
        
        if 'cpu' in task.lower():
            return [
                "Check for existing CPU monitoring tool",
                "Run the tool to get CPU usage",
                "Parse and format the results",
                "Provide human-readable summary"
            ]
        elif 'research' in task.lower() or 'compare' in task.lower():
            return [
                "Search for relevant information",
                "Fetch detailed content from top results",
                "Analyze and synthesize findings",
                "Provide comprehensive report"
            ]
        else:
            # Generic plan
            return [
                "Analyze the task requirements",
                "Gather necessary information",
                "Execute the main action",
                "Verify results"
            ]
    
    def _execute_step(self, step: str, original_task: str, previous_results: List[Dict]) -> str:
        """Execute a single step"""
        print(f"  🤖 Generating code...")
        
        # Generate code for this step
        prompt = f"""Execute this step: "{step}"

Original task: {original_task}
Previous results: {json.dumps(previous_results, indent=2)}

Generate Python code using subprocess to:
1. Call tools from {TOOLS_DIR} when possible
2. Use built-in libraries otherwise
3. Handle errors gracefully

Output ONLY executable Python code in ```python block."""
        
        try:
            result = subprocess.run(
                ['ollama', 'run', self.model, prompt],
                capture_output=True, text=True, timeout=60
            )
            
            response = result.stdout
            
            # Extract and execute code
            code = self._extract_code(response)
            if code:
                exec_result = self._execute_code(code)
                print(f"  ✅ Executed")
                return exec_result
            else:
                return "No code generated"
        
        except Exception as e:
            return f"Error: {str(e)}"
    
    def _execute_code(self, code: str) -> str:
        """Execute generated code safely"""
        import tempfile
        
        # Fix common issues
        code = code.replace('/Users/Johnson/', '/Users/Jakin/')
        code = code.replace("['3',", "['python3',")
        
        # Write to temp file
        with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False) as f:
            f.write(code)
            tmp = f.name
        
        try:
            result = subprocess.run(
                ['python3', tmp],
                capture_output=True, text=True, timeout=30,
                cwd=AGENT_DIR
            )
            
            import os
            os.unlink(tmp)
            
            if result.returncode == 0:
                return result.stdout[:1500]
            else:
                return f"Error: {result.stderr[:500]}"
        except Exception as e:
            import os
            os.unlink(tmp)
            return f"Failed: {str(e)}"
    
    def _extract_code(self, text: str) -> str:
        """Extract code from markdown"""
        match = re.search(r'```python\n(.*?)\n```', text, re.DOTALL)
        if match:
            return match.group(1)
        return None
    
    def _replan(self, task: str, results: List[Dict]) -> List[str]:
        """Create new plan based on failures"""
        # Simplified replanning
        return ["Retry with alternative approach"]
    
    def _synthesize(self, task: str, results: List[Dict]) -> str:
        """Create final answer"""
        # Combine all results into final answer
        output = "\n".join([r['result'] for r in results[-3:]])
        return f"✅ Task Complete\n\n{output[:1000]}"
